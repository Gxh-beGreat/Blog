### 核心代码

```javascript
var protection = function() {
  function loop(times) {
    if (('' + times / times)['length'] !== 1 || times % 20 === 0) {
      (function() {}['constructor']('debugger')());
    } else {
      (function() {}['constructor']('debugger')());
    }
    loop(++times);
  }
  try {
    loop(0);
  } catch (error) {}
};
protection();
```
